Algoritmo TabuadaDoNove
// Declaração de variáveis
Var
  n, dezena, unidade, resultado: inteiro

Inicio
  // Entrada de dados
  Escreva("Digite um número de 1 a 9:")
  Leia(n)

  // Processamento
  dezena <- n - 1
  unidade <- 10 - n
  resultado <- (dezena * 10) + unidade

  // Saída de dados
  Escreva("O resultado de 9 x ", n, " é ", resultado)
Fim
